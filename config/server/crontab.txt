# Renice puma and nginx to have maximum CPU priority (each every other minute).
*/2 * * * *  ps aux | grep [p]uma | awk {'print $2'} | xargs renice -n -20 -p
1-59/2 * * * *  ps aux | grep -E '[n]ginx.*process' | awk {'print $2'} | xargs renice -n -20 -p
